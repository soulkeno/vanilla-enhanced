import requests from bs4 import BeautifulSoup import time import random def get_clothing_details(clothing_name): # Base URL for the site (this will depend on the website you're scraping from) url = f'https://example.com/search?q={clothing_name}' # Replace with actual search URL headers = { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36' } try: # Send the GET request print(f"Searching for: {clothing_name}...") response = requests.get(url, headers=headers) response.raise_for_status() # Will raise an HTTPError for bad responses # Parse the page content soup = BeautifulSoup(response.text, 'html.parser') # Find the container where the clothing item details are (adjust as per the website structure) product = soup.find('div', class_='product-container') # Adjust class based on the website if not product: print(f"Couldn't find this clothing: {clothing_name}") return None # Extract product details product_name = product.find('h3', class_='product-title').text.strip() # Adjust as per website product_price = product.find('span', class_='price').text.strip() # Adjust as per website product_url = product.find('a', href=True)['href'] # Adjust as per website # You can also extract other information (like size options, images, etc.), depending on what you need product_details = { 'name': product_name, 'price': product_price, 'url': product_url } return product_details except requests.exceptions.RequestException as e: print(f"Request error: {e}") return None def main(): print("Welcome to the Clothing Search Script!") clothing_name = input("Enter the clothing name: ") # Attempt to get the product details for the given clothing name clothing_details = get_clothing_details(clothing_name) if clothing_details: print(f"\nProduct Found!") print(f"Name: {clothing_details['name']}") print(f"Price: {clothing_details['price']}") print(f"URL: {clothing_details['url']}") else: print(f"\nNo details found for '{clothing_name}'. Try another product or check the website.") if __name__ == "__main__": main()
